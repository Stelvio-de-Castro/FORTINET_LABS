CREATE VLAN ON CISCO SWITCH

conf t
hostname SW1
vlan 10
 name DMZ_VLAN10
vlan 20
 name DMZ_VLAN20
interface e0/0
 switchport trunk encapsulation dot1q
 switchport mode trunk
interface e0/3
 switchport mode access
 switchport access vlan 10
interface e0/2
 switchport mode access
 switchport access vlan 20
end
write memory


SET IP ADDRESSES ON ROUTERS, ROUTING PROTOCOL AND EACH DEFAULF GATEWAY


R10 (DMZ VLAN10 Router)
conf t
interface e0/0
ip address 172.16.10.10 255.255.255.0
no shutdown
ip route 0.0.0.0 0.0.0.0 172.16.10.1
end
write memory




R20 (DMZ VLAN20 Router)
conf t
interface e0/0
ip address 172.16.20.10 255.255.255.0
no shutdown
ip route 0.0.0.0 0.0.0.0 172.16.20.1
end
write memory
